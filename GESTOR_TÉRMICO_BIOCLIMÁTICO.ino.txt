#include <WiFi.h>
#include <WebServer.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include "DHT.h"

// --- CONFIGURAÇÃO WI-FI ---
const char* ssid = "NOME_DO_SEU_WIFI"; 
const char* password = "SENHA_DO_WIFI";

WebServer server(80);
LiquidCrystal_I2C lcd(0x27, 16, 2);

#define PIN_R 4
#define PIN_G 17
#define PIN_B 16
#define RELAY_PIN 18
#define DHTPIN_INT 13
#define DHTPIN_EXT 12
#define DHTTYPE DHT22

DHT dhtInt(DHTPIN_INT, DHTTYPE);
DHT dhtExt(DHTPIN_EXT, DHTTYPE);

// --- PÁGINA WEB (DASHBOARD) ---
void handleRoot() {
  float tI = dhtInt.readTemperature();
  float tE = dhtExt.readTemperature();
  float hE = dhtExt.readHumidity();
  
  String corStatus = (tE < (tI - 2.0) && hE < 70.0) ? "#007bff" : (hE >= 70.0 ? "#ffc107" : "#dc3545");
  String modoStr = (tE < (tI - 2.0) && hE < 70.0) ? "ECONOMIA ATIVA" : (hE >= 70.0 ? "UMIDADE ALTA" : "MODO PADRAO");

  String html = "<html><head><meta name='viewport' content='width=device-width, initial-scale=1' charset='utf-8'>";
  html += "<style>body{font-family:sans-serif; text-align:center; background:#f0f2f5; margin:0;}";
  html += "header{background:#198754; color:white; padding:15px; font-size:20px;}";
  html += ".container{padding:20px;} .card{background:white; border-radius:15px; padding:20px; margin:10px; box-shadow:0 4px 6px rgba(0,0,0,0.1); display:inline-block; min-width:150px;}";
  html += ".status{padding:15px; border-radius:10px; color:white; font-weight:bold; background:" + corStatus + ";}</style>";
  html += "<script>setInterval(function(){location.reload();}, 3000);</script></head><body>";
  html += "<header>IFCE - Gestor Termico Bioclimatico</header>";
  html += "<div class='container'>";
  html += "<div class='card'><h3>Interno</h3><h2 style='color:#333;'>" + String(tI, 1) + "C</h2></div>";
  html += "<div class='card'><h3>Externo</h3><h2 style='color:#333;'>" + String(tE, 1) + "C</h2><p>Umidade: " + String(hE, 0) + "%</p></div>";
  html += "<div style='margin-top:20px;'><div class='status'>" + modoStr + "</div></div>";
  html += "</div></body></html>";
  
  server.send(200, "text/html", html);
}

void setup() {
  Serial.begin(115200);
  lcd.init();
  lcd.backlight();
  lcd.print("Conectando WiFi");

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nConectado! IP: ");
  Serial.println(WiFi.localIP());
  
  lcd.clear();
  lcd.print("IP: ");
  lcd.setCursor(0,1);
  lcd.print(WiFi.localIP()); // Mostra o IP no LCD para você digitar no celular
  delay(3000);

  server.on("/", handleRoot);
  server.begin();

  pinMode(PIN_R, OUTPUT);
  pinMode(PIN_G, OUTPUT);
  pinMode(PIN_B, OUTPUT);
  pinMode(RELAY_PIN, OUTPUT);
  dhtInt.begin();
  dhtExt.begin();
}

void loop() {
  server.handleClient();
  
  float tInt = dhtInt.readTemperature();
  float tExt = dhtExt.readTemperature();
  float hExt = dhtExt.readHumidity();

  if (isnan(tInt) || isnan(tExt)) return;

  // Monitoramento Local LCD
  lcd.setCursor(0, 0);
  lcd.print("I:"); lcd.print(tInt, 1); lcd.print("C E:"); lcd.print(tExt, 1); lcd.print("C");

  lcd.setCursor(0, 1);
  if (tExt < (tInt - 2.0)) {
    if (hExt < 70.0) {
      digitalWrite(RELAY_PIN, HIGH);
      definirCor(0, 0, 255); // Azul
      lcd.print("MODO: ECONOMIA  ");
    } else {
      digitalWrite(RELAY_PIN, LOW);
      definirCor(255, 255, 0); // Amarelo
      lcd.print("HUMID. ALTA     ");
    }
  } else {
    digitalWrite(RELAY_PIN, LOW);
    definirCor(255, 0, 0); // Vermelho
    lcd.print("MODO: AR-COND.  ");
  }
}

void definirCor(int r, int g, int b) {
  analogWrite(PIN_R, r);
  analogWrite(PIN_G, g);
  analogWrite(PIN_B, b);
}
