#include <WiFi.h>
#include <WebServer.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include "DHT.h"

// --- CREDENCIAIS DAS REDES ---
const char* wifi_privado = "Navegamais"; 
const char* pass_privado = "TonyS@les20"; 
const char* wifi_aberto  = "IFCE-PECEM";

// Configuração do modo "Sem Internet" (Access Point)
const char* ap_nome = "PROJETO_EFICIENCIA";
const char* ap_pass = "12345678";

WebServer server(80);
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Definição de Pinos
#define PIN_R 4
#define PIN_G 2
#define PIN_B 15
#define RELE_AC 5       // Seu Relé do Ar-Condicionado
#define RELE_MOTOR 33   // Seu Relé do Motor / Ventilador
#define DHTPIN_INT 23
#define POT_EXT 34      // GPIO 34 para o Potenciômetro
#define DHTTYPE DHT22

DHT dhtInt(DHTPIN_INT, DHTTYPE);

float tI = 25.0, tE = 0, hE = 50.0; 
String ip_atual = "";

// Variáveis para controlar a alternância do LCD sem travar o código
unsigned long tempo_anterior = 0;
bool mostrar_temperaturas = true;

// ==========================================
// DASHBOARD WEB (HTML/CSS)
// ==========================================
void handleRoot() {
  String corStatus = "#dc3545";
  String modoStr = "MODO PADRAO (AC LIBERADO)";

  if (tE < (tI - 2.0)) {
    if (hE < 70.0) {
      corStatus = "#007bff";
      modoStr = "ECONOMIA ATIVA (VENTILACAO)";
    } else {
      corStatus = "#ffc107"; 
      modoStr = "SEGURANCA: UMIDADE ALTA (AC REALIMENTADO)";
    }
  }

  String html = "<html><head><meta name='viewport' content='width=device-width, initial-scale=1' charset='utf-8'>";
  html += "<style>body{font-family:Arial; text-align:center; background:#f0f2f5; margin:0;}";
  html += "header{background:#198754; color:white; padding:20px; font-size:24px; font-weight:bold;}";
  html += ".container{padding:20px;} .card{background:white; border-radius:15px; padding:25px; margin:10px; box-shadow:0 4px 10px rgba(0,0,0,0.1); display:inline-block; min-width:200px;}";
  html += ".status{padding:20px; border-radius:10px; color:white; font-size:1.2em; font-weight:bold; margin-top:20px; background:" + corStatus + ";}</style>";
  html += "<script>setInterval(function(){location.reload();}, 3000);</script></head><body>";
  html += "<header>IFCE - GESTOR TERMICO IOT</header>";
  html += "<div class='container'>";
  html += "<div class='card'><h3>Temp. Interna</h3><h2 style='font-size:3em; color:#333;'>" + String(tI, 1) + "°C</h2></div>";
  html += "<div class='card'><h3>Temp. Externa</h3><h2 style='font-size:3em; color:#333;'>" + String(tE, 1) + "°C</h2><p>Simulado via Potenciômetro</p></div>";
  html += "<div class='card'><h3>Umidade Interna</h3><h2 style='font-size:3em; color:#333;'>" + String(hE, 1) + "%</h2></div>";
  html += "<div><div class='status'>" + modoStr + "</div></div>";
  html += "</div></body></html>";
  
  server.send(200, "text/html", html);
}

void definirCor(int r, int g, int b) {
  analogWrite(PIN_R, r);
  analogWrite(PIN_G, g);
  analogWrite(PIN_B, b);
}

void conectar_wifi() {
  lcd.clear();
  lcd.print("1.Tentando Priv.");
  
  WiFi.begin(wifi_privado, pass_privado);
  int tentativas = 0;
  while (WiFi.status() != WL_CONNECTED && tentativas < 10) {
    delay(500);
    tentativas++;
  }

  if (WiFi.status() != WL_CONNECTED) {
    lcd.clear();
    lcd.print("2.Tentando Aber.");
    WiFi.begin(wifi_aberto, "");
    int tentativas_aberto = 0;
    while (WiFi.status() != WL_CONNECTED && tentativas_aberto < 10) {
      delay(500);
      tentativas_aberto++;
    }
  }

  if (WiFi.status() != WL_CONNECTED) {
    WiFi.softAP(ap_nome, ap_pass);
    ip_atual = WiFi.softAPIP().toString();
    lcd.clear();
    lcd.print("3.Modo AP Ativo");
  } else {
    ip_atual = WiFi.localIP().toString();
    lcd.clear();
    lcd.print("WiFi Conectado!");
  }
  delay(1500); 
}

// ==========================================
// SETUP E CONFIGURAÇÕES
// ==========================================
void setup() {
  Serial.begin(115200);
  
  pinMode(PIN_R, OUTPUT);
  pinMode(PIN_G, OUTPUT);
  pinMode(PIN_B, OUTPUT);
  pinMode(RELE_AC, OUTPUT);
  pinMode(RELE_MOTOR, OUTPUT);
  pinMode(POT_EXT, INPUT);

  digitalWrite(RELE_AC, HIGH);
  digitalWrite(RELE_MOTOR, HIGH);

  lcd.init();
  lcd.backlight();

  conectar_wifi();

  server.on("/", handleRoot);
  server.begin();
  dhtInt.begin();
}

// ==========================================
// LOOP PRINCIPAL
// ==========================================
void loop() {
  server.handleClient(); 
  
  // Leitura do Sensor DHT
  float t_lida = dhtInt.readTemperature();
  float h_lida = dhtInt.readHumidity();
  
  if (!isnan(t_lida)) tI = t_lida;
  if (!isnan(h_lida)) hE = h_lida; 
  
  // Leitura estável do potenciômetro
  int somaPot = 0;
  for(int i = 0; i < 10; i++) { 
    somaPot += analogRead(POT_EXT); 
    delay(2); 
  }
  int leituraPot = somaPot / 10;
  tE = map(leituraPot, 0, 4095, 150, 450) / 10.0;

  // --- LÓGICA DE EFICIÊNCIA ENERGÉTICA ---
  if (tE < (tI - 2.0)) {
    if (hE < 70.0) {
      digitalWrite(RELE_AC, HIGH);     // DESLIGA AC
      digitalWrite(RELE_MOTOR, LOW);   // LIGA o Ventilador 
      definirCor(0, 0, 255);           // Azul
    } 
    else {
      digitalWrite(RELE_AC, LOW);      // LIGA o AC
      digitalWrite(RELE_MOTOR, HIGH);  // DESLIGA o Ventilador
      definirCor(255, 120, 0);         // Amarelo
    }
  } 
  else {
    digitalWrite(RELE_AC, LOW);        // LIGA AC
    digitalWrite(RELE_MOTOR, HIGH);    // DESLIGA o Ventilador
    definirCor(255, 0, 0);             // Vermelho
  }

  // --- ATUALIZAÇÃO DO DISPLAY LCD ---
  // Linha 1 fixa com o IP
  lcd.setCursor(0, 0);
  lcd.print("IP:" + ip_atual);
  for (int i = ("IP:" + ip_atual).length(); i < 16; i++) {
    lcd.print(" ");
  }

  // Lógica de tempo para alternar a Linha 2 a cada 3000ms (3 segundos)
  unsigned long tempo_atual = millis();
  if (tempo_atual - tempo_anterior >= 3000) {
    tempo_anterior = tempo_atual;
    mostrar_temperaturas = !mostrar_temperaturas; // Inverte o estado
    lcd.setCursor(0, 1);
    lcd.print("                "); // Limpa a linha antiga rapidamente antes de reescrever
  }

  // Escreve na Linha 2 de acordo com o estado atual
  lcd.setCursor(0, 1);
  if (mostrar_temperaturas) {
    // Estado 1: Mostra as Temperaturas
    lcd.print("I:"); lcd.print(tI, 1); lcd.print("C E:"); lcd.print(tE, 1); lcd.print("C ");
  } else {
    // Estado 2: Mostra a Umidade
    lcd.print("Umidade: "); lcd.print(hE, 1); lcd.print("%   ");
  }

  delay(20); 
}
